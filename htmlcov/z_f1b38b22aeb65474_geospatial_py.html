21def lat_lon_to_nta(lat: pd.Series, lon: pd.Series) -> pd.Series: 

22 """Vectorized borough-bucket assignment for synthetic NTA-style zone IDs. 

23 

24 Best-effort proxy until real NTA boundary geometry is available for 

25 point-in-polygon. Inputs must be aligned numeric Series of equal length. 

26 """ 

27 lat_arr = lat.to_numpy(dtype=float) 

28 lon_arr = lon.to_numpy(dtype=float) 

29 

30 conditions = [ 

31 (lat_arr > 40.78) & (lon_arr < -73.82), 

32 (lat_arr < 40.65) & (lon_arr < -74.1), 

33 (lon_arr < -73.95) & (lat_arr >= 40.63) & (lat_arr <= 40.73), 

34 lon_arr < -73.75, 

35 ] 

36 borough = np.select(conditions, ["BX", "SI", "BK", "QN"], default="MN") 

37 bucket = np.abs(lat_arr * 10).astype(int) % 100 

38 bucket_str = np.char.zfill(bucket.astype(str), 2) 

39 return pd.Series(np.char.add(borough.astype(str), bucket_str), index=lat.index)
