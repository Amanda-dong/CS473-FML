77def _borough_key(nta_id: str) -> str: 

78 prefix2 = nta_id[:2].upper() 

79 if prefix2 in _BOROUGH_PARAMS: 

80 return prefix2 

81 prefix_lower = nta_id[:2].lower() 

82 return _BOROUGH_PREFIX_MAP.get(prefix_lower, "MN")

85def _build_synthetic_acs(limit: int) -> pd.DataFrame: 

86 """Generate plausible NTA-level ACS data using borough-level seeds.""" 

87 rows: list[dict] = [] 

88 for nta_id in _ZONE_IDS: 

89 borough = _borough_key(nta_id) 

90 params = _BOROUGH_PARAMS[borough] 

91 for year in _YEARS: 

92 seed = abs(hash(nta_id + str(year))) % 100_000 

93 rng = np.random.default_rng(seed) 

94 noise = rng.uniform(0.85, 1.15) 

95 growth = 1.0 + 0.02 * (year - 2018) # 2% annual income growth 

96 rows.append( 

97 { 

98 "year": year, 

99 "nta_id": nta_id, 

100 "median_income": round(params["median_income"] * growth * noise), 

101 "population": round(params["population"] * noise), 

102 "rent_burden": round( 

103 min( 

104 0.85, 

105 max(0.10, params["rent_burden"] * rng.uniform(0.90, 1.10)), 

106 ), 

107 3, 

108 ), 

109 "_synthetic": True, 

110 } 

111 ) 

112 df = pd.DataFrame(rows) 

113 return df.head(limit)

116def run_placeholder_etl() -> pd.DataFrame:

120def _load_local() -> pd.DataFrame: 

121 """Load ACS data from a local CSV specified by env var.""" 

122 path_str = os.environ.get("ACS_DATA_PATH", "") 

123 if not path_str: 

124 raise RuntimeError("etl_acs: ACS_DATA_PATH env var not set") 

125 path = Path(path_str) 

126 if not path.is_file(): 

127 raise FileNotFoundError(f"etl_acs: ACS_DATA_PATH={path} does not exist") 

128 logger.info("etl_acs: loading from local file %s", path) 

129 return pd.read_csv(path) 

130 

131 

132def run_etl(limit: int = 50000) -> pd.DataFrame: 

133 """Load real ACS data; falls back to borough-seeded synthetic data if unavailable.""" 

134 try: 

135 df = _load_local() 

136 if df.empty: 

137 raise RuntimeError("etl_acs: local file returned empty frame") 

138 return df.head(limit) 

139 except (RuntimeError, FileNotFoundError) as exc: 

140 logger.warning( 

141 "etl_acs: real data unavailable (%s) — using synthetic fallback", exc 

142 ) 

143 return _build_synthetic_acs(limit)
